#include <iostream>
#include <stdlib.h>
#include <time.h>

using namespace std;

int main(){
int suma = 0;
    srand(time(NULL));
    for (int i=0;i<100;i++){
        int x=rand() % 100 + 1;
        
        if (x%2==0) //dzielenie modulo
        suma = x + suma;
        
        cout << x << endl;
    }

cout << "suma liczb parzystych = " << suma << endl;

    system("pause");
    return 0;
}
